nginx-开启缓存

发表于 2017-03-01 
//图片
location ~* ^.+\.(ico|gif|jpg|jpeg|png)$ { 
    access_log   off; 
    expires      30d;
}
//css js 
location ~* ^.+\.(css|js|txt|xml|swf|wav)$ {
    access_log   off;
    expires      24h;
}
//html
location ~* ^.+\.(html|htm)$ {
    expires      1h;
}
//字体
location ~* ^.+\.(eot|ttf|otf|woff|svg)$ {
    access_log   off;
    expires max;
}
